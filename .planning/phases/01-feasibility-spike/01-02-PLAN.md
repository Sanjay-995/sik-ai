  static func estimateScale(session: ARSession, footPixel: CGPoint, headPixel: CGPoint, userHeightCm: Double) -> ScaleResult
  Primary path:
    - Check session.currentFrame?.camera.trackingState == .normal
    - Build ARRaycastQuery(origin: ..., direction: ..., allowing: .estimatedPlane, alignment: .horizontal)
    - If raycast hits: compute footDepth = simd_distance(cameraPos, hit.worldTransform.columns.3.xyz)
    - focalLength = Double(frame.camera.intrinsics.columns.0.x)

    - Return ScaleResult(source: .arkit)
  Fallback path (ARKit unavailable or limited):

    - metersPerPixel = (userHeightCm / 100.0) / pxSpan
    - Return ScaleResult(source: .userHeight)

  Pure computation — no Vision or ARKit imports.
  struct Shot { let poseResult: PoseResult; let scaleResult: ScaleResult; let segMask: CVPixelBuffer?; let view: CaptureView }
  enum CaptureView { case front, side }
  struct CircumferenceMeasurements: Codable — all nine fields (waist_cm, hip_cm, chest_cm, neck_cm, shoulders_cm, left_upper_arm_cm, right_upper_arm_cm, left_thigh_cm, right_thigh_cm) as Double? (optional)
  static func compute(front: Shot, side: Shot, params: HeuristicParams.Values) -> CircumferenceMeasurements
    For waist, hip, chest:
      1. Determine anatomical image-y from joint positions using params fractions (e.g., for chest: imageY = shoulderJoint.y + params.chestYFromShoulder * (hipJoint.y - shoulderJoint.y))
      2. Get widthFront from front.segMask via SegmentService.bodyWidthMeters at that y (fallback to keypoint distance if segMask nil)
      3. Get widthSide from side.segMask similarly
      4. a = widthFront / 2.0; b = widthSide / 2.0
      5. C_meters = Double.pi * (3*(a+b) - sqrt((3*a+b)*(a+3*b)))  // Ramanujan
      6. return C_meters * 100.0  // convert to cm
    Neck, shoulders, arms, thighs: compute only if all required joints present at confidence >= params.minJointConfidence; otherwise nil (D-01 tiered scope)
    NO magic numbers — every constant comes from params argument (HeuristicParams.locked)

Final locked content:
```swift
import CryptoKit
import Foundation

/// SPIKE PARAMETER LOCK — Do not edit this file during the data-collection window.
/// Any edit invalidates the spike dataset. See Resources/README.md for rationale.
enum HeuristicParams {
    struct Values: Codable {
        // Anatomical y-fractions (fraction of shoulder->hip joint span)
        let chestYFromShoulder: Double       // 0.20: ~20% down from shoulder = axillary/fullest chest level
        let waistYNarrowestBandLow: Double   // 0.40: begin narrowest-waist search band
        let waistYNarrowestBandHigh: Double  // 0.65: end narrowest-waist search band
        let hipYOffsetBelowHipJoint: Double  // -0.05: 5% below hip joint = greater trochanter level

        let usePlainRamanujan: Bool          // true: use Ramanujan first approximation C = pi*(3(a+b)-sqrt((3a+b)(a+3b)))

        // Pose acceptance gates
        let minJointConfidence: Double       // 0.4
        let minRequiredJoints: Int           // 15 of 17 joints must meet minJointConfidence

        // Same-session outlier trimming for 5-shot stability sub-test
        let sessionMedianTrim: Double        // 0.20: drop top+bottom 10% of rapid-fire session results
    }

    static let locked = Values(
        chestYFromShoulder: 0.20,
        waistYNarrowestBandLow: 0.40,
        waistYNarrowestBandHigh: 0.65,
        hipYOffsetBelowHipJoint: -0.05,
        usePlainRamanujan: true,
        minJointConfidence: 0.4,
        minRequiredJoints: 15,
        sessionMedianTrim: 0.20
    )
}

extension HeuristicParams.Values {
    /// SHA256 hash of the JSON-encoded locked parameters.
    /// Embedded in every JSONL scan record as heuristic_params_hash.
    var hash: String {
        let encoder = JSONEncoder()
        encoder.outputFormatting = .sortedKeys
        guard let data = try? encoder.encode(self) else { return "hash-error" }
        let digest = SHA256.hash(data: data)
        return digest.map { String(format: "%02x", $0) }.joined()
    }
}
```

## LOCKED Heuristic Parameters (D-10)

These values are committed to git and must NOT be changed during data collection. Any change invalidates the dataset and requires restarting data collection.

| Parameter | Value | Rationale |
|-----------|-------|-----------|
| chestYFromShoulder | 0.20 | Chest is ~20% down from the shoulder toward the hip joint. Maps to axillary level (fullest chest) per ISO 8559-1. |
| waistYNarrowestBandLow | 0.40 | Natural waist is in the middle third of shoulder-to-hip span. Search starts at 40% down. ISO 8559-1: natural waist = smallest horizontal circumference between ribs and iliac crest. |
| waistYNarrowestBandHigh | 0.65 | Search ends at 65% down. |
| hipYOffsetBelowHipJoint | -0.05 | Greater trochanter (max hip circumference) is ~3-6% below the hip joint center. -0.05 is the midpoint estimate. |
| usePlainRamanujan | true | Use Ramanujan first approximation: C = pi*(3(a+b) - sqrt((3a+b)(a+3b))). No additional bias correction — that would itself be a tunable. |
| minJointConfidence | 0.40 | Joints below this confidence threshold are discarded from circumference calculation. |
| minRequiredJoints | 15 | At least 15 of 17 Vision 3D joints must meet minJointConfidence for a scan to be accepted. |
| sessionMedianTrim | 0.20 | For the 5-shot stability sub-test: drop top+bottom 10% of readings before computing session mean. |

## Spike Protocol Summary

- 3 testers required — one per BMI bracket: <22 (lean), 22-27 (normal), >27 overweight (ideally >32 obese). See .planning/phases/01-feasibility-spike/01-RECRUITMENT-PLAN.md.
