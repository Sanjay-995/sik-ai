Gate tiers per D-03:
  PASS: worst inter-session SD across all (bracket x measurement) cells <= 2.0 cm
  SOFT_PASS: 2.0 < worst <= 3.0 cm (document specific weakness)
  HARD_REPLAN: worst > 3.0 cm (stop, reconvene)

Repeatability coefficient: CR = within-subject SD * 2.77 (Vaz et al. 2013)
  PASS gate at SD=2.0 -> CR = 5.54 cm (95% bound on same-body difference)

```python
#!/usr/bin/env python3
"""
SikAI Spike Harness — Off-device reproducibility analysis
Usage: python analyze_spike.py --input path/to/scans.jsonl [--output-dir ./results]
"""
import argparse
import json
import os
import sys
from pathlib import Path
import pandas as pd
import numpy as np
from scipy import stats

# ─── Parse args ────────────────────────────────────────────────────────────────
parser = argparse.ArgumentParser()
parser.add_argument("--input", default="../scans.jsonl", help="Path to scans.jsonl")
parser.add_argument("--output-dir", default="./results", help="Output directory for CSVs")
args = parser.parse_args()

input_path = Path(args.input)
output_dir = Path(args.output_dir)
output_dir.mkdir(parents=True, exist_ok=True)

# ─── Load data ────────────────────────────────────────────────────────────────
if not input_path.exists():
    print(f"ERROR: {input_path} not found", file=sys.stderr)
    sys.exit(1)

rows = []
with open(input_path) as f:
    for line in f:
        line = line.strip()
        if not line:
            continue
        data = json.loads(line)
        # Flatten nested fields
        circ = data.get("computed_circumferences_cm", {}) or {}
        tape = data.get("tape_measurements_cm", {}) or {}
        rows.append({
            "tester_id": data.get("tester_id"),
            "tester_bmi_bracket": data.get("tester_bmi_bracket"),
            "session_id": data.get("session_id"),
            "scan_id": data.get("scan_id"),
            "timestamp": data.get("timestamp"),
            "condition_label": data.get("condition_label"),
            "scale_source": data.get("scale_source"),
            "pose_confidence_min": data.get("pose_confidence_min"),
            "waist_cm": circ.get("waist"),
            "hip_cm": circ.get("hip"),
            "chest_cm": circ.get("chest"),
            "tape_waist_cm": tape.get("waist"),
            "tape_hip_cm": tape.get("hip"),
            "tape_chest_cm": tape.get("chest"),
        })

df = pd.DataFrame(rows)
print(f"Loaded {len(df)} scan rows from {input_path}")
print(f"Testers: {sorted(df['tester_id'].dropna().unique())}")
print(f"BMI brackets: {sorted(df['tester_bmi_bracket'].dropna().unique())}")
print(f"Conditions: {sorted(df['condition_label'].dropna().unique())}")

# ─── Analysis 1: Inter-session SD per tester per measurement (GATE METRIC D-04) ─
measurements = ["waist_cm", "hip_cm", "chest_cm"]
session_means = (df
    .groupby(["tester_id", "tester_bmi_bracket", "session_id"])[measurements]
    .mean()
    .reset_index())

# Per (tester, bracket, measurement): SD across session means
inter_sd = (session_means
    .melt(id_vars=["tester_id", "tester_bmi_bracket", "session_id"],
          value_vars=measurements, var_name="measurement", value_name="cm")
    .groupby(["tester_id", "tester_bmi_bracket", "measurement"])["cm"]
    .agg(sd_cm="std", n_sessions="count", mean_cm="mean")
    .reset_index())

# Per BMI bracket (worst case SD per bracket per measurement)
per_bracket_sd = (inter_sd
    .groupby(["tester_bmi_bracket", "measurement"])["sd_cm"]
    .agg(max_sd_cm="max", mean_sd_cm="mean", n_testers="count")
    .reset_index())

per_bracket_sd["repeatability_coefficient_cm"] = per_bracket_sd["max_sd_cm"] * 2.77

inter_sd.to_csv(output_dir / "inter_session_sd.csv", index=False)
per_bracket_sd.to_csv(output_dir / "per_bracket_sd.csv", index=False)

# ─── Analysis 2: Accuracy vs tape measure ─────────────────────────────────────
tape_cols = {"waist_cm": "tape_waist_cm", "hip_cm": "tape_hip_cm", "chest_cm": "tape_chest_cm"}
acc_rows = []
for meas, tape_col in tape_cols.items():
    subset = df[[
        "tester_id", "tester_bmi_bracket", "session_id", "scan_id",
        "condition_label", meas, tape_col
    ]].dropna(subset=[meas, tape_col])
    if subset.empty:
        continue
    subset = subset.copy()
    subset["error_cm"] = subset[meas] - subset[tape_col]
    subset["abs_error_cm"] = subset["error_cm"].abs()
    subset["measurement"] = meas
    acc_rows.append(subset)

if acc_rows:
    accuracy_df = pd.concat(acc_rows, ignore_index=True)
    accuracy_summary = (accuracy_df
        .groupby(["tester_id", "tester_bmi_bracket", "measurement"])
        .agg(mean_error_cm=("error_cm", "mean"),
             mae_cm=("abs_error_cm", "mean"),
             n=("abs_error_cm", "count"))
        .reset_index())
    accuracy_summary.to_csv(output_dir / "accuracy_vs_tape.csv", index=False)
else:
    print("WARNING: No tape measurements found in dataset — skipping accuracy analysis")

# ─── Analysis 3: Condition split (controlled vs solo) ────────────────────────
condition_session_means = (df
    .assign(condition_group=lambda x: x["condition_label"].apply(
        lambda c: "solo" if str(c).startswith("solo") else "controlled"))
    .groupby(["tester_id", "tester_bmi_bracket", "condition_group", "session_id"])[measurements]
    .mean()
    .reset_index())

condition_sd = (condition_session_means
    .melt(id_vars=["tester_id", "tester_bmi_bracket", "condition_group", "session_id"],
          value_vars=measurements, var_name="measurement", value_name="cm")
    .groupby(["tester_id", "tester_bmi_bracket", "condition_group", "measurement"])["cm"]
    .agg(sd_cm="std", n_sessions="count")
    .reset_index())

condition_sd.to_csv(output_dir / "condition_split.csv", index=False)

# ─── Gate decision (D-03) ─────────────────────────────────────────────────────
# Use per-bracket worst-case SD across all bracket x measurement cells
worst_sd = per_bracket_sd["max_sd_cm"].max()

if worst_sd <= 2.0:
    tier = "PASS"
    tier_note = "All (bracket x measurement) cells have inter-session SD <= 2.0 cm. Proceed to Phase 2."
elif worst_sd <= 3.0:
    tier = "SOFT_PASS"
    soft_failures = per_bracket_sd[per_bracket_sd["max_sd_cm"] > 2.0]
    tier_note = (
        f"Inter-session SD exceeds 2.0 cm in {len(soft_failures)} (bracket x measurement) cell(s). "
        "Proceed to Phase 2 with documented weakness. Phase 3 capture-flow tightening is the planned mitigation.\n"
        "Cells exceeding 2.0 cm:\n" + soft_failures.to_string(index=False)
    )
else:
    tier = "HARD_REPLAN"
    hard_failures = per_bracket_sd[per_bracket_sd["max_sd_cm"] > 3.0]
    tier_note = (
        f"Inter-session SD exceeds 3.0 cm in {len(hard_failures)} (bracket x measurement) cell(s). "
        "STOP. Reconvene on stack and scope before Phase 2.\n"
        "Cells exceeding 3.0 cm:\n" + hard_failures.to_string(index=False)
    )

gate_text = f"""
GATE TIER: {tier}
Worst inter-session SD across all (bracket x measurement) cells: {worst_sd:.2f} cm
Repeatability coefficient (CR = SD * 2.77) for worst cell: {worst_sd * 2.77:.2f} cm

{tier_note}

D-03 thresholds:
  PASS: worst SD <= 2.0 cm
  SOFT_PASS: 2.0 < worst SD <= 3.0 cm
  HARD_REPLAN: worst SD > 3.0 cm
"""

with open(output_dir / "gate_decision.txt", "w") as f:
    f.write(gate_text)

print("\n" + "="*60)
print("GATE DECISION:")
print(gate_text)
print("="*60)
print(f"\nOutputs written to {output_dir}/")
print("  inter_session_sd.csv   — primary gate metric (per tester per measurement)")
print("  per_bracket_sd.csv     — aggregated per BMI bracket")
print("  accuracy_vs_tape.csv   — bias vs tape measure (secondary)")
print("  condition_split.csv    — controlled vs solo split (secondary)")
print("  gate_decision.txt      — tier verdict (machine-readable)")
```
